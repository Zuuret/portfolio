Knowledge Distillation Loss

def compute_distillation_loss(student_logits, teacher_logits, labels, 
                               alpha=0.7, temperature=4.0):
    # KD Loss: soft targets from teacher
    teacher_probs = F.softmax(teacher_logits / temperature, dim=-1)
    kd_loss = F.kl_div(
        F.log_softmax(student_logits / temperature, dim=-1),
        teacher_probs,
        reduction='batchmean'
    ) * (temperature ** 2)
    
    # CE Loss: hard targets (ground truth)
    ce_loss = F.cross_entropy(student_logits, labels)
    
    # Combined loss
    total_loss = alpha * kd_loss + (1 - alpha) * ce_loss
    return total_loss

Multi-Round Distillation Loop

def multi_round_distillation(teacher_model, student_model, 
                              train_loader, num_rounds=10):
    results = []
    
    for round_num in range(num_rounds):
        optimizer = AdamW(student_model.parameters(), lr=2e-5)
        
        for batch in train_loader:
            student_logits = student_model(batch['input_ids'])
            teacher_logits = teacher_model(batch['input_ids'])
            
            loss = compute_distillation_loss(
                student_logits, teacher_logits, batch['labels'],
                alpha=0.7, temperature=4.0
            )
            loss.backward()
            optimizer.step()
        
        # Evaluate
        acc = evaluate(student_model, val_loader)
        results.append(acc)
        print(f"Round {round_num + 1}: Accuracy = {acc:.4f}")
    
    return results

Dataset avec Tokenization

class VulnerabilityDataset(Dataset):
    def __init__(self, data, tokenizer, max_length=512):
        self.data = data
        self.tokenizer = tokenizer
        self.max_length = max_length
    
    def __len__(self):
        return len(self.data)
    
    def __getitem__(self, idx):
        code = self.data[idx]['code']
        label = self.data[idx]['label']
        
        encoding = self.tokenizer(
            code,
            max_length=self.max_length,
            truncation=True,
            padding='max_length',
            return_tensors='pt'
        )
        
        return {
            'input_ids': encoding['input_ids'].squeeze(),
            'attention_mask': encoding['attention_mask'].squeeze(),
            'labels': torch.tensor(label, dtype=torch.long)
        }
