void Server::run()
{
    std::vector<pollfd> fds;
    
    while (g_running)
    {
        // Build pollfd array
        fds.clear();
        for (size_t i = 0; i < _clients.size(); ++i)
            fds.push_back(_clients[i].getPollFd());
        
        int poll_ret = poll(fds.data(), fds.size(), TIMEOUT_MS);
        if (poll_ret < 0)
            throw std::runtime_error("poll error");
        
        // Process events
        for (size_t i = 0; i < fds.size(); ++i)
        {
            if (fds[i].revents & POLLIN)
                _clients[i].handleRead();
            if (fds[i].revents & POLLOUT)
                _clients[i].handleWrite();
            if (fds[i].revents & (POLLERR | POLLHUP))
                _clients[i].close();
        }
    }
}

        _parseHeaders(header_part);

        _body_offset = header_end + 4;
    }
    
    if (_state == BODY)
    {
        int body_len = std::stoi(_headers["Content-Length"]);
        std::string body = data.substr(_body_offset);
        
        if (body.length() >= body_len)
        {
            _body = body.substr(0, body_len);
            _state = COMPLETE;
            return true;
        }
    }
    return false;
}
