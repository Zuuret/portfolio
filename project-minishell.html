void execute_pipe(t_command *cmd1, t_command *cmd2)
{
    int pipefd[2];
    pid_t pid1, pid2;
    
    if (pipe(pipefd) == -1)
        exit_error("pipe failed");
    
    pid1 = fork();
    if (pid1 == 0)
    {
        dup2(pipefd[1], STDOUT_FILENO);
        close(pipefd[0]);
        close(pipefd[1]);
        execute_command(cmd1);
        exit(EXIT_FAILURE);
    }
    
    pid2 = fork();
    if (pid2 == 0)
    {
        dup2(pipefd[0], STDIN_FILENO);
        close(pipefd[0]);
        close(pipefd[1]);
        execute_command(cmd2);
        exit(EXIT_FAILURE);
    }
    
    close(pipefd[0]);
    close(pipefd[1]);
    waitpid(pid1, NULL, 0);
    waitpid(pid2, NULL, 0);
}

t_token *tokenize(char *input)
{
    t_token *tokens;
    int i;
    int j;
    char in_quote;
    
    tokens = malloc(sizeof(t_token) * (ft_strlen(input) + 1));
    i = 0;
    j = 0;
    in_quote = 0;
    
    while (input[i])
    {
        if ((input[i] == '\'' || input[i] == '\"') && !in_quote)
            in_quote = input[i++];
        else if (input[i] == in_quote)
        {
            in_quote = 0;
            i++;
        }
        else if (input[i] == ' ' && !in_quote)
        {
            tokens[j++].type = TOKEN_SPACE;
            i++;
        }
        else
        {
            tokens[j++].type = TOKEN_ARG;
            tokens[j].value = extract_arg(&input[i], &in_quote);
            i += ft_strlen(tokens[j].value);
        }
    }
    tokens[j].type = TOKEN_EOF;
    return (tokens);
}

void signal_handler(int sig)
{
    if (sig == SIGINT)
    {
        write(1, "\n", 1);

    }
}
